+ `class`를 이용한 값 확인 방법

```js
class Timer {
  constructor(props) {
    this.role = 1000;
    this.sec = props.sec;
    this.interval = null;
    this.count = 0;
  }
  getSec() {
    return this.count;
  }
  start() {
    const self = this;
    self.count = self.sec;
    self.interval = setInterval(() => {
      console.log("동작");
      self.count--;
    }, self.role);
  }
  stop() {
    const self = this;
    clearInterval(self.interval);
  }
}

const timerSetting = {
  sec: 300,
};
const timer = new Timer(timerSetting);
timer.start();
```
